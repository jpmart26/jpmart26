<script>
function calculate() {
  var h = document.getElementById('hour').value;
  var m = document.getElementById('minute').value;
  var err = document.getElementById('err');

  if (h === '' || m === '') {
    err.textContent = '請輸入時同分';
    return;
  }
  h = parseInt(h);
  m = parseInt(m);
  if (h < 0 || h > 23 || m < 0 || m > 59) {
    err.textContent = '時間範圍：時 0-23，分 0-59';
    return;
  }
  err.textContent = '';

  var total = h * 60 + m;

  function fmt(mins) {
    mins = ((mins % 1440) + 1440) % 1440;
    var hh = String(Math.floor(mins / 60)).padStart(2, '0');
    var mm = String(mins % 60).padStart(2, '0');
    return hh + ':' + mm;
  }

  document.getElementById('leaveTime').textContent = fmt(total - 150);
  document.getElementById('prepTime').textContent = fmt(total - 210);
}
</script>
